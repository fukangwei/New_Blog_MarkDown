&emsp;&emsp;代码如下：

``` cpp
#include<iostream>
#include <vector>
​
using namespace std;
​
void init_vector ( vector <int> &vecTest );
void show_vector ( const vector <int> &vecTest );
​
int main() {
    vector<int> vec; /* 声明一个vector */
    init_vector ( vec ); /* 初始化vector */
    show_vector ( vec ); /* 显示vector */
}
​
/* 初始化容器 */
void init_vector ( vector <int> &vecTest ) {
    for ( int i = 0; i < 10; i++ ) {
        /* push_back方法用于在vector的末尾附加元素 */
        vecTest.push_back ( i );
    }
}
​
/* 显示容器中的所有数据 */
void show_vector ( const vector <int> &vecTest ) {
    vector<int>::const_iterator it = vecTest.begin();
​
    while ( it < vecTest.end() ) {
        cout << *it << endl;
        it++;
    }
}
```

其他容器类或者`String`类也可以使用这种方法作为函数的参数。
