&emsp;&emsp;代码如下：<!--more-->

``` cpp
#include <iostream>
#include <fstream>

const char *filename = "out.txt";

int main () {
    long l, m;
    ifstream in ( filename, ios::in | ios::binary );
    l = in.tellg();
    in.seekg ( 0, ios::end );
    m = in.tellg();
    in.close();
    cout << "size of " << filename;
    cout << " is " << ( m - l ) << " bytes.\n";
    return 0;
}
```

执行结果：

``` cpp
size of out.txt is 40 bytes.
```
